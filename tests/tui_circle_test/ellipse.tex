    \begin{flalign*}
        \frac{\left(x - h\right)^2}{h^2} + \frac{\left(y - v\right)^2}{v^2} &= 1 & \\
        \frac{\left(x - h\right)^2}{h^2} &= 1 - \frac{\left(y - v\right)^2}{v^2} & \\
        \left(x - h\right)^2 &= h^2 \left[ 1 - \frac{\left(y - v\right)^2}{v^2} \right] & \\
        \because y &\in [v, 2v] & \\
        v \leq y &\leq 2v & \\
        v - v \leq y - v &\leq 2v - v & \\
        0 \leq y - v &\leq v & \\
        0 \leq (y - v)^2 &\leq v^2 & \\
        0 \leq \frac{(y - v)^2}{v^2} &\leq \frac{v^2}{v^2} & \\
        0 \leq \frac{(y - v)^2}{v^2} &\leq 1 & \\
        -1 \leq -\frac{(y - v)^2}{v^2} &\leq 0 & \\
        0 \leq 1 - \frac{(y - v)^2}{v^2} &\leq 1 & \\
        1 - \frac{(y - v)^2}{v^2} &\geq 0 & \\
        x - h &= \pm \sqrt{h^2 \left[ 1 - \frac{\left(y - v\right)^2}{v^2} \right]} & \\
        x - h &= \pm \sqrt{h^2} \sqrt{ 1 - \frac{\left(y - v\right)^2}{v^2} } & \\
        x &= \pm h \sqrt{ 1 - \frac{\left(y - v\right)^2}{v^2} } + h &
    \end{flalign*}
